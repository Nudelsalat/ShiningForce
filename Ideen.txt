private var _movementMap = currentMovementMap // which is an array array.

4,4,3
private MovementGrid MoveFunction(x, y, movement){
	var movementGrid = new Pos(x,y)

	var restMovement = movement - _movementMap[x-1][y];
	if(restMovement >= 0){
		movementGrid.Add(MoveFunction(x-1,y, restMovement));
	}
	
	restMovement = movement - _movementMap[x][y-1];
	if(restMovement >= 0){
		movementGrid.Add(MoveFunction(x,y-1, restMovement));
	}
	
	restMovement = movement - _movementMap[x+1][y];
	if(restMovement >= 0){
		movementGrid.Add(MoveFunction(x+1,y, restMovement));
	}

	restMovement = movement - _movementMap[x][y+1];
	if(restMovement >= 0){
		movementGrid.Add(MoveFunction(x,y+1, restMovement));
	}

	return movementGrid;
}
